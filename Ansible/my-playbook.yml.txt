---
  - name: Config Web VM with docker
    hosts: webservers
    become: true
    tasks:
    - name: docker.io
      apt:
        name: docker.io
        state: present
    - name: pip3
      apt:
        name: python3-pip
        state: present
    - name: Docker module
      pip:
        name: docker
        state: present
    - name: download and launch a docker web container
      docker_container:
        name: dvwa
        image: cyberxsecurity/dvwa
        state: started
        restart_policy: always
        published_ports: 80:80
